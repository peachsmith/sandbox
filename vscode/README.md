### Debugging C/C++ on Windows using MinGW
task.json:
```json
{
    "tasks": [
        {
            "type": "cppbuild",
            "label": "debug-build",
            "command": "mingw32-make",
            "args": [
                "debug"
            ],
            "options": {
                "cwd": "${workspaceFolder}\\<path_to_working_directory>"
            },
            "problemMatcher": [
                "$gcc"
            ],
            "group": {
                "kind": "build",
                "isDefault": true
            },
            "detail": "Task generated by Debugger."
        }
    ],
    "version": "2.0.0"
}
```

launch.json:
```json
{
    "configurations": [
        {
            "name": "C/C++: mingw32-make build and debug active file",
            "type": "cppdbg",
            "request": "launch",
            "program": "${workspaceFolder}\\<path_to_executable>",
            "args": [],
            "stopAtEntry": false,
            "cwd": "${workspaceFolder}\\<path_to_working_directory>",
            "environment": [],
            "externalConsole": false,
            "MIMode": "gdb",
            "miDebuggerPath": "gdb.exe",
            "setupCommands": [
                {
                    "description": "Enable pretty-printing for gdb",
                    "text": "-enable-pretty-printing",
                    "ignoreFailures": true
                },
                {
                    "description": "Set Disassembly Flavor to Intel",
                    "text": "-gdb-set disassembly-flavor intel",
                    "ignoreFailures": true
                }
            ],
            "preLaunchTask": "debug-build"
        }
    ],
    "version": "2.0.0"
}
```
